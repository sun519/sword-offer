function FirstNotRepeatingChar(str) {
    var hash = {};
    for(var i = 0; i < str.length; i++) {
        if(hash[str[i]] === undefined) {
            hash[str[i]] = i;
        }else {
            hash[str[i]] = false;
        }
    }
    for(var key in hash) {
        if(hash[key] !== false) {
            return hash[key];
        }
    }
    return -1;
}
